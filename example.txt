  <script>
    function toggleAccordion(element) {
      const content = element.nextElementSibling;
      const icon = element.querySelector(".icon");
      const allContent = document.querySelectorAll(".accordion-content");
      const allIcons = document.querySelectorAll(".icon");

      // Close other sections
      allContent.forEach((el) => {
        if (el !== content) el.classList.remove("open");
      });
      allIcons.forEach((el) => {
        if (el !== icon) el.classList.remove("rotate");
      });

      // Toggle clicked section
      content.classList.toggle("open");
      icon.classList.toggle("rotate");
    }

    function scrollContent(button, amount) {
      const content = button.parentElement;
      content.scrollBy({ top: amount, behavior: "smooth" });
    }
  </script>
